  <button onclick="calculateTrip()" style="width: 100%; padding: 1rem; background: #1e40af; color: white; border: none; border-radius: 8px; font-weight: 600; cursor: pointer; margin-top: 1rem;">Calculate Total Cost</button>

<script>
function calculateTrip() {
  const daily = parseFloat(document.getElementById('dailyBudget').value);
  const days = parseFloat(document.getElementById('days').value);
  const flight = parseFloat(document.getElementById('flight').value);
  
  const accommodation = daily * days;
  const total = accommodation + flight;
  const perDay = total / days;
  
  document.getElementById('tripResult').innerHTML = `
    <div style="background: white; padding: 1.5rem; border-radius: 8px; border: 2px solid #1e40af;">
      <p style="margin: 0 0 0.5rem 0;"><strong>Daily Expenses:</strong> $${accommodation.toLocaleString()}</p>
      <p style="margin: 0 0 0.5rem 0;"><strong>Flight:</strong> $${flight.toLocaleString()}</p>
      <p style="margin: 0 0 1rem 0; padding-top: 0.5rem; border-top: 1px solid #e2e8f0;"><strong>Total Trip Cost:</strong> $${total.toLocaleString()}</p>
      <p style="margin: 0; font-size: 1.2rem; color: #1e40af;"><strong>True Daily Cost:</strong> $${perDay.toFixed(2)}/day</p>
    </div>
  `;
}
</script>
